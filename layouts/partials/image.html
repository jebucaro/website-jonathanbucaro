{{- /* Responsive image renderer. Resolves `image` (a resource, or a string tried against page resources, then assets/, then treated as a plain URL), generates WebP width variants (never upscaling; the original width is always included, re-encoded), and emits a complete with srcset/sizes/width/height. Passthrough (plain , no processing): SVG, GIF, external URLs, data: URIs, unresolvable paths. Params (dict): image resource or string (required) page page context; required to resolve page-resource strings alt alt text (required) sizes HTML sizes attribute (default "100vw") widths candidate srcset widths (default: 400 700 1000 1400) loading "lazy" (default) or "eager" fetchpriority optional, e.g. "high" for LCP images class optional extra class names title optional title attribute */ -}} {{- $input := .image -}} {{- $page := .page -}} {{- $alt := .alt -}} {{- $sizes := .sizes | default "100vw" -}} {{- $widths := .widths | default (slice 400 700 1000 1400) -}} {{- $loading := .loading | default "lazy" -}} {{- $fetchpriority := .fetchpriority -}} {{- $title := .title -}} {{- /* Resolve string input: page resources, then assets/, else plain URL */ -}} {{- $res := "" -}} {{- $rawURL := "" -}} {{- if eq (printf "%T" $input) "string" -}} {{- $rawURL = $input -}} {{- if not (or (hasPrefix $input "http") (hasPrefix $input "//") (hasPrefix $input "data:")) -}} {{- with $page -}}{{- $res = .Resources.GetMatch $input -}}{{- end -}} {{- if not $res -}} {{- $res = resources.Get (strings.TrimPrefix "/" $input) -}} {{- end -}} {{- end -}} {{- else -}} {{- $res = $input -}} {{- end -}} {{- /* Only raster formats Hugo can process; SVG/GIF pass through */ -}} {{- $processable := false -}} {{- with $res -}} {{- $processable = in (slice "png" "jpeg" "webp" "tiff" "bmp") .MediaType.SubType -}} {{- end -}} {{- $classes := slice -}} {{- with .class -}}{{- $classes = $classes | append . -}}{{- end -}} {{- if eq $loading "lazy" -}}{{- $classes = $classes | append "fade-img" -}}{{- end -}} {{- if $processable -}} {{- $candidates := slice -}} {{- range $widths -}} {{- if lt . $res.Width -}}{{- $candidates = $candidates | append . -}}{{- end -}} {{- end -}} {{- $candidates = $candidates | append $res.Width -}} {{- $variants := slice -}} {{- range $candidates -}} {{- $variants = $variants | append ($res.Resize (printf "%dx webp" .)) -}} {{- end -}} {{- $largest := index $variants (sub (len $variants) 1) -}} {{- /* src fallback for no-srcset clients: variant closest to 700px */ -}} {{- $fallback := index $variants 0 -}} {{- range $variants -}} {{- if lt (math.Abs (sub .Width 700)) (math.Abs (sub $fallback.Width 700)) -}} {{- $fallback = . -}} {{- end -}} {{- end -}} {{- $srcset := slice -}} {{- range $variants -}} {{- $srcset = $srcset | append (printf "%s %dw" .RelPermalink .Width) -}} {{- end -}} {{- else -}} {{- $url := $rawURL -}} {{- with $res -}}{{- $url = .RelPermalink -}}{{- end -}} {{- if $url -}} {{- else -}} {{- warnf "partials/image.html: could not resolve image %v" $input -}} {{- end -}} {{- end -}}
