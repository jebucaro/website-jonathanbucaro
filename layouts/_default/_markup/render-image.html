{{- $src := .Destination -}} {{- $alt := .Text -}} {{- $title := .Title -}} {{- /* Check if relative path (not http/https or absolute) */ -}} {{- if not (or (hasPrefix $src "http") (hasPrefix $src "//") (hasPrefix $src "/")) -}} {{- /* Try to get from page resources */ -}} {{- $resource := .Page.Resources.GetMatch $src -}} {{- if $resource -}} {{- /* Use resource's relURL */ -}} {{- $src = $resource.RelPermalink -}} {{- else -}} {{- /* Fallback: Build path from permalink, removing language prefix for non-English languages */ -}} {{- $pagePath := .Page.RelPermalink -}} {{- /* For Spanish pages, strip /es/ prefix to access shared resources */ -}} {{- if eq .Page.Language.Lang "es" -}} {{- $pagePath = strings.TrimPrefix "/es/" $pagePath -}} {{- $pagePath = printf "/%s" $pagePath -}} {{- end -}} {{- $src = printf "%s%s" $pagePath $src -}} {{- end -}} {{- end -}}
